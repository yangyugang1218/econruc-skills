\sectionframe{Introduction}

\begin{frame}{Research motivation}
  \begin{wideitemize}
    \item Why does this economic margin matter?
    \item What puzzle, institution, or empirical fact makes the question nontrivial?
    \item Why should the audience care before seeing the model or results?
  \end{wideitemize}
\end{frame}

\begin{frame}{Research question}
  \begin{wideitemize}
    \item What is the exact question?
    \item What comparison or counterfactual is needed to answer it?
    \item Why is the answer not already known?
  \end{wideitemize}
\end{frame}

\begin{frame}{Research contribution}
  \begin{wideitemize}
    \item \blue{Substantive:} what do we learn about the economic mechanism?
    \item \blue{Design/data/model:} what lets the paper answer the question?
    \item \blue{Literature/policy:} what debate does the paper move?
  \end{wideitemize}
\end{frame}

\begin{frame}{Preview of main findings}
  \begin{wideitemize}
    \item \blue{Finding 1:} state the main quantitative or theoretical result.
    \item \blue{Mechanism:} explain the key economic channel.
    \item \blue{Implication:} state what changes about policy or theory.
  \end{wideitemize}
\end{frame}

\begin{frame}{Outline}
  \begin{wideitemize}
    \item Introduction and setting
    \item Model or identification strategy
    \item Data and main results
    \item Mechanisms and robustness
    \item Conclusion
  \end{wideitemize}
\end{frame}

\begin{frame}{A conceptual contrast}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.46\textwidth}
      \conceptbox{Benchmark}{%
        \begin{itemize}\setlength{\itemsep}{0.35em}
          \item What would standard theory predict?
          \item What would happen under frictionless competition?
        \end{itemize}
      }
    \end{column}
    \begin{column}{0.46\textwidth}
      \conceptbox{This paper}{%
        \begin{itemize}\setlength{\itemsep}{0.35em}
          \item What friction or strategic force changes the prediction?
          \item What evidence distinguishes the mechanism?
        \end{itemize}
      }
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Conclusion}
  \begin{wideitemize}
    \item This paper studies \blue{[question]}.
    \item It shows \blue{[main answer]} using \blue{[design/model/data]}.
    \item The result matters because \blue{[economic implication]}.
  \end{wideitemize}
\end{frame}
